#+BEGIN_SRC python
def readb64(data):
    nparr = np.frombuffer(base64.b64decode(data), np.uint8)
    return cv2.imdecode(nparr, cv2.IMREAD_COLOR)
#+END_SRC

** Read image

#+BEGIN_SRC python
